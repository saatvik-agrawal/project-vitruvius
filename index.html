    <script>

        function findWorkshop() {

            var input = document.getElementById("workshopInput").value; // Get selected requirement

            var result = document.getElementById("result"); // Get result display area

            

            // Object storing workshop recommendations based on selected input

            var workshops = {

                "community": "Community Engagement with Engineers: Learn how engineers can engage with local communities beyond technology.",

                "discussion": "Fishbowl Method: A structured discussion method for debating different perspectives.",

                "visual": "Graph Visualization: Represent community preferences using visual charts.",

                "modelling": "Modeling Workshop: Use creative materials to design community solutions.",

                "photography": "A Week with a Camera: Participants document their surroundings to discuss neighborhood improvements.",

                "mapping": "Mapping Activity: Identify valued and problematic spaces in a community.",

                "survey": "Visual Preference Survey: Collect insights about preferred public spaces.",

                "design": "Collaborative Design Patterns: Explore real-world design problems using case studies."

            };

            

            // Check if the selected input matches a workshop

            if (input in workshops) {

                result.innerHTML = "<strong>Recommended Workshop:</strong> " + workshops[input];

                result.style.display = "block"; // Show the result area

            } else {

                result.style.display = "none"; // Hide the result area if no selection

            }

        }

    </script>
